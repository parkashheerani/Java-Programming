public class Characters {

    public static void main(String[] args) {

        char c;

        for(c = 'A'; c <= 'Z'; ++c)
            System.out.print(c + " ");
    }
}

public class Characters {

    public static void main(String[] args) {

        char c;

        for(c = 'a'; c <= 'z'; ++c)
            System.out.print(c + " ");
    }
}
